<!-- 전화번호 입력 폼 -->
<form id="phoneNumberForm">
    <label for="phoneNumber">Phone Number:</label>
    <input type="text" id="phoneNumber" name="phoneNumber" required>
    <button type="submit">Submit</button>
</form>

<script>
document.getElementById('phoneNumberForm').addEventListener('submit', function (event) {
    event.preventDefault();

    const phoneNumber = document.getElementById('phoneNumber').value;

    // 서버에 전화번호를 전송하고 추천 메뉴를 받아옴
    fetch(`http://localhost:4000/api/recommendations/${phoneNumber}`)
        .then(response => response.json())
        .then(data => {
            if (data.success) {
                const recommendation = data.recommendation;
                if (recommendation) {
                    // 추천 메뉴를 화면에 표시
                    alert(`추천 메뉴: ${recommendation.menu_title}`);
                    
                    // 메뉴 선택 화면을 보여주는 함수 호출
                    showMenuSelection();
                } else {
                    alert('이전 주문 이력이 없습니다.');
                }
            } else {
                console.error('Error fetching recommendation:', data.message);
            }
        })
        .catch(error => {
            console.error('Error fetching recommendation:', error);
        });
});

// 메뉴 선택 화면을 보여주는 함수
function showMenuSelection() {
    // 실제로 메뉴 선택 화면을 보여주는 로직 추가
    const menuSelectionDiv = document.createElement('div');
    menuSelectionDiv.id = 'menuSelection';
    menuSelectionDiv.innerHTML = `
        <h2>Menu Selection</h2>
        <ul>
            <li><button onclick="selectMenu(1)">Menu 1</button></li>
            <li><button onclick="selectMenu(2)">Menu 2</button></li>
            <!-- 추가 메뉴 버튼을 추가할 수 있습니다. -->
        </ul>
    `;

    // 기존 내용을 지우고 새로운 메뉴 선택 화면을 추가
    document.body.innerHTML = '';
    document.body.appendChild(menuSelectionDiv);
}

// 메뉴 선택 시 동작하는 함수
function selectMenu(menuId) {
    alert(`선택한 메뉴: ${menuId}`);
    // 선택한 메뉴에 대한 추가 동작을 구현할 수 있습니다.
}
</script>
